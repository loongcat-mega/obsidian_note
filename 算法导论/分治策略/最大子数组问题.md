寻找非空数组A的最大非空连续子数组，我们称这样的连续子数组为最大子数组(maximum subarray)


## 非递归求解

```c++
int maxProfit(vector<int>& prices) {

        int max=0;

        int n=prices.size()-1;

        int max_pri=prices[n];

        for(int i=n;i>=0;i--)

        {

            if(prices[i]>max_pri)

                max_pri=prices[i];

            if(max_pri-prices[i]>max)

                max=max_pri-prices[i];

        }

        return max;

    }
```

## 分治法求解

int max_cross_subarray(vector<int>&v,int l,int r)  
{  
    int left_sum=-100000009,right_sum=-100000009;  
    int mid=l+(r-l)/2;  
    int Max=0;  
    for(int i=mid;i>=l;i--)  
    {  
        Max+=v[i];  
        if(Max>left_sum)  
            left_sum=Max;  
    }  
    Max=0;  
    for(int i=mid+1;i<=r;i++)  
    {  
        Max+=v[i];  
        if(Max>right_sum)  
            right_sum=Max;  
    }  
    return max left_sum+right_sum;  
} 
int max_subarray(vector<int>&v,int l,int r)  
{  
    if(l==r)  
        return v[l];  
    int m=l+(r-l)/2;  
    int 
    left_sum= max_subarray(v,l,m),
    cross_sum= max_cross_subarray(v,l,r),  
    right_sum= max_subarray(v,m+1,r);  
    return max(left_sum,max(cross_sum,right_sum));  
}
