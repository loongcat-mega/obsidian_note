bool StackOrder()
{
	Stack<int>s(N);
	bool *flag=new bool [N];
	for(int i=0;i<N;i++)
		flag[i]=false;
	for(int i=0;i<N:i++)
		if(!flag[i])
			while(1)
			{
				if(!flag[i])
				{
					Visit(i);
					flag[i]=true;
					s.Push(i);
				}
				int j=0;
				for(;j<N;j++)
					if(Adj[i][j]!=MAX&&!flag[j])
					{
						i=j;//从j开始搜索
						break;
					}
				if(j==N)
				{
					s.Pop();
					if(s.IsEmpty())
						return true;
					i=s.Top();
				}
			}
	return true;
}
