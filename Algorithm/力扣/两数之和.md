vector<int> twoSum(vector<int>& nums, int target)  
{  
    unordered_map<int,int>hashmap;  
    int n=(int)nums.size();  
    for(int i=n-1;~i;i--)  
    {  
        if(hashmap.find(target-nums[i])!=hashmap.end())  
            return {i,hashmap[target-nums[i]]};  
        hashmap[nums[i]]=i;  
    }  
    return {};  
}
