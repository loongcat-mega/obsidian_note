vector<int> HDAmul(vector<int>v1, vector<int> v2)
{
	vector<int>vt(v1.size()+v2.size());
	for (int i = 0; i < v1.size(); i++)
		for (int j = 0; j < v2.size(); j++)
		{
			vt[i + j] += v1[i] * v2[j];
			vt[i + j+1] +=vt[i+j]/10;
			vt[i + j] %= 10;
		}
	while (vt.size() > 1 && vt.back() == 0)
		vt.pop_back();
	return vt;
}
